local function safeBox(part, color, size, t)
	if not part or not part:IsA("BasePart") then return end
	local box = part:FindFirstChild("ESPBox")
	if not box then
		box = Instance.new("BoxHandleAdornment")
		box.Name = "ESPBox"
		box.AlwaysOnTop = true
		box.ZIndex = 0
		box.Adornee = part
		box.Parent = part
	end
	box.Size = size
	box.Color3 = color or Color3.fromRGB(255,255,0)
	box.Transparency = t
end

local function createText(parent, text, rgb, transparency, size)
	if parent:FindFirstChild("BillboardText") then return end

	local billboard = Instance.new("BillboardGui")
	billboard.Name = "BillboardText"
	billboard.Size = size
	billboard.Adornee = parent
	billboard.AlwaysOnTop = true
	billboard.LightInfluence = 0
	billboard.StudsOffset = Vector3.new(0, 3, 0)
	billboard.Parent = parent

	local label = Instance.new("TextLabel")
	label.Size = UDim2.new(1, 0, 1, 0)
	label.BackgroundTransparency = 1
	label.TextScaled = true
	label.Text = text
	label.TextColor3 = rgb or Color3.fromRGB(255, 255, 255)
	label.TextTransparency = transparency or 0
	label.Font = Enum.Font.SciFi
	label.Parent = billboard

	return billboard
end

local function distance(a, b)
    return (a - b).Magnitude
end

local function checkMonsters()
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end
    local hrp = char.HumanoidRootPart

    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("Model") or obj:IsA("BasePart") then
            local name = string.lower(obj.Name)
            if string.find(name, "monster") and not string.find(name, "monster2") then
                local pos
                if obj:IsA("BasePart") then
                    pos = obj.Position
                elseif obj.PrimaryPart then
                    pos = obj.PrimaryPart.Position
                end
                if pos and distance(hrp.Position, pos) <= 100 then
                    obj:Destroy()
                end
            end
        end
    end
end

local function doScanOnce()
	if not Workspace or not Workspace:FindFirstChild("rooms") then return end
	for _, desc in ipairs(Workspace.rooms:GetDescendants()) do
		local name = (desc.Name and desc.Name:lower()) or ""
		if desc:IsA("Model") and name == "battery" then
			for _, p in ipairs(desc:GetDescendants()) do
				if p:IsA("BasePart") then
					safeBox(p, Color3.fromRGB(255,255,0), Vector3.new(1,1,1), 0.6)
					createText(p, "Battery", Color3.fromRGB(255,255,0), 0.6, UDim2.new(0, 90, 0, 20))
				end
			end
		
		elseif desc:IsA("Model") and name == "hidetable" then
		    createText(desc, "Table", Color3.fromRGB(200,50,255), 0.6, UDim2.new(0, 120, 0, 30))

		elseif desc:IsA("Model") and name == "hidelocker" then
			local hasJack = false
			for _, v in ipairs(desc:GetDescendants()) do
				if v.Name:lower() == "jack" then
					hasJack = true
					break
				end
			end

			if desc:FindFirstChild("hitbox") then
				local billboard = desc.hitbox:FindFirstChild("BillboardText")
				if not hasJack then
					safeBox(desc.hitbox, Color3.fromRGB(0,255,0), Vector3.new(3,6,3), 0.7)
					if not billboard then
						billboard = createText(desc.hitbox, "Locker", Color3.fromRGB(0,255,0), 0.7, UDim2.new(0, 130, 0, 30))
					end
					if billboard and billboard:FindFirstChildOfClass("TextLabel") then
						local label = billboard:FindFirstChildOfClass("TextLabel")
						label.Text = "Locker"
						label.TextColor3 = Color3.fromRGB(0,255,0)
					end
				else
					safeBox(desc.hitbox, Color3.fromRGB(255,0,0), Vector3.new(3,6,3), 0.7)
					if not billboard then
						billboard = createText(desc.hitbox, "Shells", Color3.fromRGB(255,0,0), 0, UDim2.new(0, 140, 0, 40))
					end
					if billboard and billboard:FindFirstChildOfClass("TextLabel") then
						local label = billboard:FindFirstChildOfClass("TextLabel")
						label.Text = "Shells"
						label.TextColor3 = Color3.fromRGB(255,0,0)
						if label.Text == "Shells" and mm5 then
						desc:Destroy()
						end
					end
				end
			end
        
		elseif desc:IsA("BasePart") and name == "door" and (desc.Parent and desc.Parent.Name:lower() ~= "hidelocker") then
			safeBox(desc, Color3.fromRGB(0,150,150), desc.Size, 0.8)
		end
	end
end

local function updateMonsterESP(inst, labelName)
    if not inst:IsA("BasePart") and not inst.PrimaryPart then return end
    
    local root = inst:IsA("Model") and inst.PrimaryPart or inst
    if not root then return end

    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end
    local hrp = char.HumanoidRootPart
    
    local distance = (hrp.Position - root.Position).Magnitude
    local text = string.format("%s\n[%.0f studs]", labelName, distance)

    safeBox(root, Color3.fromRGB(255,0,0), Vector3.new(10,10,10), 0.5)

    local billboard = root:FindFirstChild("BillboardText")
    if not billboard then
        billboard = createText(root, text, Color3.fromRGB(255,0,0), 0, UDim2.new(0, 160, 0, 50))
    end

    if billboard and billboard:FindFirstChildOfClass("TextLabel") then
        billboard:FindFirstChildOfClass("TextLabel").Text = text
    end
end

local function doMonsterESP()
    for _, inst in ipairs(Workspace:GetChildren()) do
        local name = (inst.Name and inst.Name:lower()) or ""
        if name == "monster" then
            updateMonsterESP(inst, "Monster")
        elseif name == "monster2" then
            updateMonsterESP(inst, "Monster2")
        elseif name == "???" then
            updateMonsterESP(inst, "A-666")
        end
    end
end
